   import express from "express";
import axios from "axios";
const app = express();
const PORT = process.env.PORT || 3000;

app.get("/dailyVideo", async (req, res) => {
  try {
    const topic = "latest gadget news";
    const scriptResp = await axios.post(
      "https://api.openai.com/v1/chat/completions",
      {
        model: "gpt-4-turbo",
        messages: [
          { role: "system", content: "You are Milo White, the AI dog reviewer." },
          { role: "user", content: Write a 45-second tech short about ${topic}. Include fun tone and pros / cons. }
        ]
      },
      { headers: { Authorization: Bearer ${process.env.OPENAI_KEY} } }
    );
    res.json({ success: true, script: scriptResp.data.choices[0].message.content });
  } catch (e) {
    res.status(500).json({ success: false, error: e.message });
  }
});

app.listen(PORT, () => console.log(Server running on port ${PORT}));
